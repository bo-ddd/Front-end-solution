```html
<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>Document</title>
  <style>
    div {
      background-color: var(--bg-color);
    }
  </style>
</head>
<body>
  <div>hello world</div>
  <button>换肤</button>
</body>
<script>
  class Theme {
    constructor() {
        this._theme = 'light'; // 初始主题设置为'light'
        this.init(); // 初始化时设置默认主题
    }

    init() {
        // 根据当前主题设置样式
        this.setTheme(this._theme);
    }

    setTheme(theme) {
        // 移除之前的主题样式（如果需要）
        this.removeTheme();

        // 设置新主题的样式
        if (theme === 'light') {
            // 添加针对light主题的样式
            // 例如，通过修改CSS变量或添加/移除class
            document.documentElement.style.setProperty('--bg-color', 'white');
            // ... 其他样式设置
        } else if (theme === 'dark') {
            // 添加针对dark主题的样式
            document.documentElement.style.setProperty('--bg-color', 'black');
            // ... 其他样式设置
        }

        // 更新内部主题状态
        this._theme = theme;
    }

    removeTheme() {
        // 这里可以根据需要重置或移除之前设置的样式
        // 例如，通过移除特定的class或重置CSS变量
    }

    tab() {
        // 切换主题
        this.setTheme(this._theme === 'light' ? 'dark' : 'light');
    }
}

// 创建Theme实例
const theme = new Theme();
  
document.querySelector('button').onclick = function() {
  theme.tab()
}
</script>
</html>
```
